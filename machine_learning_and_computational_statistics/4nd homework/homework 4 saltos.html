# Generate data
N = 5000

X = []
Y = []
for i in range(0, 100):
    x, y = np.random.multivariate_normal(mean, cov, N).T
    
    X.append(x)
    Y.append(y)

X = np.array(X)
Y = np.array(Y)

theta_set=[]
for i in range(0, 100):
    X_new = X[i].T
    XTXinv = (np.dot(X_new.T,X_new) ** (-1))
    theta_est = np.dot(XTXinv,X_new.T)
    theta_est = np.dot(theta_est,Y[number])
    theta_set.append(theta_est)

fig1 = plt.figure()
ax1 = fig1.add_subplot(111)

for i in range(0, 100):
    x = np.linspace(1,2,10)
    g = theta_set[number]*x+1
    ax1.plot(x, g, '-b')

g = 0.5*x+1
ax1.plot(x, g, '-g', label='g=0.5x+1')
plt.title('Estimates of theta for N=5000')
plt.xlabel('x', color='#1C2833')
plt.ylabel('g(x)', color='#1C2833')
plt.legend(loc='upper left')
plt.grid()
plt.show()
